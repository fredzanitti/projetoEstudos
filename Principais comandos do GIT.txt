git config --lit												Mostra todas as configurações realizadas

git commit -m "mensagem"										Comita tudo que está pendente
git commit -am "mensagem"										Comita tudo que está pendente e sobe para o servidor remoto
git log															Mostra as alterações de todos os usuários
git log --author="Fred"											Mostra as alterações de um usuário específico
git shortlog													Mostra a quantidade de alterações que um usuário realizou
git shortlog --graph											Mostra as alterações de usuários em modo gráfico
git log --graph
git show														Mostra as alterações realizadas num determinado arquivo
git diff														Mostra as diferenças entre os arquivos antes de realizar o commit
